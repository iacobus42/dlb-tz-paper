Data Load and Processing

Load the data from the network drive using the correct path for argon vs. local work.

if (Sys.info()["sysname"] == "Darwin") {
  shared_dir <- "/Volumes"
} else {
  shared_dir <- "/Shared"
}
model_data <- read_rds(glue::glue("{shared_dir}/lss_jsimmeri/LBD/treated_model_data.rds"))

outcomes <- model_data %>%
  select(enrolid, drug, develops_pd, survival_time)

And load the matching data:

tz_tam_matches <- read_rds(glue::glue("{shared_dir}/lss_jsimmeri/LBD/matches_tz_tam_2rx.rds")) %>%
  mutate(enrolid = as.numeric(enrolid))
tz_5ari_matches <- read_rds(glue::glue("{shared_dir}/lss_jsimmeri/LBD/matches_tz_5ari_2rx.rds")) %>%
  mutate(enrolid = as.numeric(enrolid))
tam_5ari_matches <- read_rds(glue::glue("{shared_dir}/lss_jsimmeri/LBD/matches_tam_5ari_2rx.rds")) %>%
  mutate(enrolid = as.numeric(enrolid))

With the matches and outcomes, build the matched outcome files:

outcomes_tz_tam <- inner_join(outcomes, tz_tam_matches, by = c("enrolid"))
outcomes_tz_5ari <- inner_join(outcomes, tz_5ari_matches, by = c("enrolid"))
outcomes_tam_5ari <- inner_join(outcomes, tam_5ari_matches, by = c("enrolid"))

Base Model and Schoenfeld Residuals

Fit the base model, same as fit in the main analysis, and use the Schoenfeld residuals to get an estimate of the time-varying hazard ratio:

cph_tz_tam <- coxph(Surv(survival_time, develops_pd) ~ drug, 
      data = outcomes_tz_tam, 
      cluster = pair_id,
      robust = TRUE)
cph_tz_5ari <- coxph(Surv(survival_time, develops_pd) ~ drug, 
      data = outcomes_tz_5ari, 
      cluster = pair_id,
      robust = TRUE)
cph_tam_5ari <- coxph(Surv(survival_time, develops_pd) ~ drug, 
      data = outcomes_tam_5ari, 
      cluster = pair_id,
      robust = TRUE)

Step Function

Event counts by period:

outcomes_tz_tam %>%
  mutate(
    period = case_when(
      survival_time <= 365 ~ "0-1",
      survival_time <= 2 * 365 ~ "1-2",
      survival_time <= 3 * 365 ~ "2-3",
      survival_time <= 4 * 365 ~ "3-4",
      survival_time <= 5 * 365 ~ "4-5",
      survival_time <= 7.5 * 365 ~ "5-7.5",
      survival_time <= 10 * 365 ~ "7.5-10",
      survival_time > 10 ~ "10+"
    )
  ) |>
  group_by(
    drug, period
  ) |>
  summarize(
    n = sum(develops_pd)
  ) |>
  pivot_wider(
    names_from = "drug",
    values_from = "n"
  )

Fit with a time-varying effect but use a step function for each year of follow-up (first year, second year, and so on to 10th or later year).

cph_step_tz_tam <- survSplit(Surv(survival_time, develops_pd) ~ ., 
          data = outcomes_tz_tam,
          cut = c(1, 2, 3, 4, 5, 7.5, 10) * 365,
          episode = "tgroup") %>%
  coxph(Surv(tstart, survival_time, develops_pd) ~ treatment:strata(tgroup),
        data = ., cluster = pair_id, robust = TRUE)

cph_step_tz_5ari <- survSplit(Surv(survival_time, develops_pd) ~ .,
